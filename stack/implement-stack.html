Time Complexities of operations on the stack:

push(), pop(), isEmpty() and peek() all take O(1) time. We do not run any loop in any of these operations.

 

Code:

#define MAX 1000

 

class Stack {

    int top;

 

public:

    int st[MAX]; // Maximum size of Stack

 

    Stack() { top = -1; }

    bool push(int x);

    int pop();

    int peek();

    bool isEmpty();

};

bool Stack::push(int x){

    if(top>=MAX-1){

        cout<<"Stack Overflow\n";

        return false;

    }

    else{

        st[++top]=x;

        cout<<x<<" pushed into the stack\n";

        return true;

    }

}

int Stack::pop(){

    if(isEmpty()){

        cout<<"Stack Underflow\n";

        return 0;

    }

    else{

        return st[top--];

    }

}

int Stack::peek(){

    if(isEmpty()){

        cout<<"Stack Underflow\n";

        return 0;

    }

    else{

        return st[top];

    }

}

bool Stack::isEmpty(){

    return (top<0);

}

Implementing Stack Using Linked List:

 

struct Node{

    int data;

    Node *next;

};

Node *top;

void push(int x){

    Node *temp = new Node;

   

    if(!temp){     //if there is no memory in heap

        cout<<"stack overflow\n";

        return;

    }

    temp->data=x;

    temp->next=top;

    top=temp;

}

bool isEmpty(){

    return top==NULL;

}

int pop(){

    Node *temp;

    if(isEmpty()){        //if stack is empty

        cout<<"stack overflow\n";

        return 0;

    }

    int x= top->data;

    temp=top;         //moving ahead and skippin the top

    temp= temp->next;

    top=temp;

    return x;

    free(temp);

}

int peek(){

    if(isEmpty()){        //if stack is empty

        cout<<"stack overflow\n";

        return 0;

    }

    return (top->data);

}

void display(){

    Node *temp;

    if(isEmpty()){

    temp=top;

        cout<<temp->data<<" -> ";

    }

}
